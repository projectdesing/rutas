public class Conductor {
    public String tipo;

    public String getTipo() {
        return tipo;
    }

    public void setTipo(String tipo) {
        this.tipo = tipo;
    }
    
    public Conductor consultar(String nombre){
        Iterable<Conductor> Conductor = null;
        for (Conductor conductor : Conductor) {
            if (conductor.getTipo().equals(nombre)) {
                return conductor;
            }
        }   
        String conductor = "No encontrado";
        return conductor;
    }
    
    public String eliminar(String nombre){
        for (Conductor conductor : Conductor) {
        if (conductor.getTipo().equals(nombre)) {
            return "conductor elimnado";
        }
        }
        return "El conductor ha sido eliminado";
    }
}
